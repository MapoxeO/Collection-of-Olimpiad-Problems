\task {
    \newtaskref{linal} \newtaskref{series}
} {
    Найти произведение
    $$
        \prod_{a \in A} \begin{bmatrix}
            \frac{1}{a^2} & 1 \\
            1 & \frac{1}{a^2}
        \end{bmatrix}
    $$
    \par в двух случаях:
    \begin{enumerate}
        \item $A=\mathbb{P} - \mbox{множество всех простых чисел}$;
        \item $A=\mathbb{N} - \mbox{множество натуральных чисел}$.
    \end{enumerate}
} {
    РЕШЕНИЕ
} {
    $1)\;\frac{3}{2\pi^2} \begin{bmatrix}
        7 & 3 \\
        3 & 7 \\
    \end{bmatrix};$
    $2)\;\frac{\sinh{\pi}}{2\pi} \begin{bmatrix}
        1 & 1 \\
        1 & 1 \\
    \end{bmatrix}.$
}
